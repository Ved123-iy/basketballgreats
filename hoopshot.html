function shootFromDrag(){ if(!ball) spawnBall(); if(!dragStart || !dragNow) return; const dx = (dragStart.x - dragNow.x); const dy = (dragStart.y - dragNow.y); // EASY MODE: stronger shots even with small drags const power = Math.min(32, Math.max(14, Math.hypot(dx, dy) / 7)); // Aim where you drag, BUT add a little auto-aim help toward the hoop const aimAngle = Math.atan2(dy, dx); const hoopAngle = Math.atan2((dragStart.y - rim.y), (rimCenterX() - dragStart.x)); const assist = 0.35; // higher = easier (0.2–0.5 is good) const angle = aimAngle * (1 - assist) + hoopAngle * assist; ball.vx = Math.cos(angle) * power; ball.vy = Math.sin(angle) * power * -1; shots++; shotsEl.textContent = shots; } function rimCenterX(){ return rim.x + rim.w/2; }
